\frame{
	\frametitle{Assymetric cryptography}
	$\textnormal{Enc}$ and $\textnormal{Dec}$ are separate functions,
	each parametrized by a key.
	
	For a valid key pair $(pk, sk)$,
	$\textnormal{Dec}_{sk}^{-1} = \textnormal{Enc}_{pk}$.


	\begin{enumerate}[<+->]
		\item Alice generates "public key" and "secret key"
		\item Alice sends the public key to Bob
		\item Bob encrypts the message using public key
		\item Bob sends the ciphertext to Alice
		\item Alice decrypts it using the secret key.
	\end{enumerate}
}
